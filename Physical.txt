#include <SPI.h>
#include <nRF24L01.h>
#include <RF24.h>

// ขา NRF24L01 สำหรับ ESP32
#define CE_PIN   4
#define CSN_PIN  5

// ขา Ultrasonic
#define TRIG_PIN 13
#define ECHO_PIN 12

// ขา Motor หุ่นคุม (L298N)
#define MOTOR_IN1 26
#define MOTOR_IN2 27

// ขาปุ่มกด
#define BTN_K1 32 // หมุนมอเตอร์ไปข้างหน้า
#define BTN_K2 33 // (ไม่ได้ใช้ในเงื่อนไขนี้)
#define BTN_K3 25 // ส่งค่าระยะ
#define BTN_K4 14 // หมุนมอเตอร์ไปข้างหลัง

RF24 radio(CE_PIN, CSN_PIN);
const byte address[6] = "00001"; // Address ต้องตรงกันทั้ง 2 ฝั่ง

void setup() {
  Serial.begin(115200);

  // Setup Motor
  pinMode(MOTOR_IN1, OUTPUT);
  pinMode(MOTOR_IN2, OUTPUT);
  digitalWrite(MOTOR_IN1, LOW);
  digitalWrite(MOTOR_IN2, LOW);

  // Setup Buttons
  pinMode(BTN_K1, INPUT_PULLUP);
  pinMode(BTN_K2, INPUT_PULLUP);
  pinMode(BTN_K3, INPUT_PULLUP);
  pinMode(BTN_K4, INPUT_PULLUP);

  // Setup Ultrasonic
  pinMode(TRIG_PIN, OUTPUT);
  pinMode(ECHO_PIN, INPUT);

  // Setup Radio
  if (!radio.begin()) {
    Serial.println("NRF24 Hardware not responding!");
    // while (1) {} // เอาออกเพื่อให้โค้ดยังทำงานส่วนอื่นได้แม้ NRF มีปัญหา
  }
  radio.openWritingPipe(address);
  radio.setPALevel(RF24_PA_MIN);
  radio.stopListening();
}

void loop() {
  // --- ส่วนควบคุมมอเตอร์หุ่นคุม (Manual Control) ---
  if (digitalRead(BTN_K1) == LOW) {
    // กด K1 หมุนไปข้างหน้า
    digitalWrite(MOTOR_IN1, HIGH);
    digitalWrite(MOTOR_IN2, LOW);
  } 
  else if (digitalRead(BTN_K4) == LOW) {
    // กด K4 หมุนไปข้างหลัง
    digitalWrite(MOTOR_IN1, LOW);
    digitalWrite(MOTOR_IN2, HIGH);
  } 
  else {
    // ไม่กด ให้หยุด
    digitalWrite(MOTOR_IN1, LOW);
    digitalWrite(MOTOR_IN2, LOW);
  }

  // --- ส่วนวัดระยะทาง (วัดตลอดเวลา) ---
  float currentDistance = getDistance();
  
  // --- ส่วนส่งค่า (ส่งเมื่อกด K3) ---
  if (digitalRead(BTN_K3) == LOW) {
    Serial.print("Sending Distance: ");
    Serial.println(currentDistance);
    
    bool report = radio.write(&currentDistance, sizeof(currentDistance));
    
    if (report) {
      Serial.println("Sent successfully");
    } else {
      Serial.println("Send failed");
    }
    
    delay(200); // ดีเลย์ป้องกันการส่งรัวเกินไปขณะกดแช่
  }

  delay(50); // Loop delay เล็กน้อย
}

float getDistance() {
  digitalWrite(TRIG_PIN, LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN, LOW);
  
  long duration = pulseIn(ECHO_PIN, HIGH);
  // คำนวณเป็น cm
  return duration * 0.034 / 2;
}

#include <SPI.h>
#include <nRF24L01.h>
#include <RF24.h>

// ขา NRF24L01 สำหรับ Nano
#define CE_PIN   7 
#define CSN_PIN  8

// ขา Motor Driver (L298N)
// สมมติ: IN1/2 คือล้อซ้าย, IN3/4 คือล้อขวา
#define IN1 2 // ซ้ายเดินหน้า
#define IN2 3 // ซ้ายถอยหลัง
#define IN3 4 // ขวาเดินหน้า
#define IN4 5 // ขวาถอยหลัง

RF24 radio(CE_PIN, CSN_PIN);
const byte address[6] = "00001";

float receivedDistance = 0.0;
unsigned long lastReceiveTime = 0;

void setup() {
  Serial.begin(9600);

  pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT);
  pinMode(IN4, OUTPUT);
  stopCar();

  if (!radio.begin()) {
    Serial.println("NRF24 Hardware not responding!");
  }
  radio.openReadingPipe(0, address);
  radio.setPALevel(RF24_PA_MIN);
  radio.startListening(); // เป็นตัวรับ
}

void loop() {
  if (radio.available()) {
    radio.read(&receivedDistance, sizeof(receivedDistance));
    lastReceiveTime = millis();
    Serial.print("Recv: "); Serial.println(receivedDistance);
    
    controlCar(receivedDistance);
  }

  // Safety: ถ้าไม่ได้รับค่าเกิน 1 วินาที ให้หยุดรถ
  if (millis() - lastReceiveTime > 1000) {
    stopCar();
  }
}

void controlCar(float dist) {
  // เงื่อนไข 1: 3.5cm - 6.5cm -> ล้อขวาหมุนไปข้างหน้า
  if (dist >= 3.5 && dist <= 6.5) {
    rightForwardOnly(); 
  }
  // เงื่อนไข 2: 8.5cm - 11.5cm -> ล้อทั้งคู่หมุนไปข้างหน้า
  else if (dist >= 8.5 && dist <= 11.5) {
    moveForward();
  }
  // เงื่อนไข 3: 13.5cm - 16.5cm -> ล้อทั้งคู่หมุนไปข้างหลัง
  else if (dist >= 13.5 && dist <= 16.5) {
    moveBackward();
  }
  // เงื่อนไข 4: 18.5cm - 21.5cm -> ล้อซ้ายหมุนไปข้างหน้า
  else if (dist >= 18.5 && dist <= 21.5) {
    leftForwardOnly();
  }
  // นอกเหนือเงื่อนไข -> หยุด
  else {
    stopCar();
  }
}

// ฟังก์ชันควบคุมมอเตอร์
void moveForward() {
  // ซ้ายเดินหน้า, ขวาเดินหน้า
  digitalWrite(IN1, HIGH); digitalWrite(IN2, LOW);
  digitalWrite(IN3, HIGH); digitalWrite(IN4, LOW);
}

void moveBackward() {
  // ซ้ายถอย, ขวาถอย
  digitalWrite(IN1, LOW); digitalWrite(IN2, HIGH);
  digitalWrite(IN3, LOW); digitalWrite(IN4, HIGH);
}

void rightForwardOnly() {
  // ซ้ายหยุด, ขวาเดินหน้า
  digitalWrite(IN1, LOW); digitalWrite(IN2, LOW);
  digitalWrite(IN3, HIGH); digitalWrite(IN4, LOW);
}

void leftForwardOnly() {
  // ซ้ายเดินหน้า, ขวาหยุด
  digitalWrite(IN1, HIGH); digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW); digitalWrite(IN4, LOW);
}

void stopCar() {
  digitalWrite(IN1, LOW); digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW); digitalWrite(IN4, LOW);
}
